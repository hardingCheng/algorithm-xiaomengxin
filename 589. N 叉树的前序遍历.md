## 解析

### 方法 1

```js
/**
 * // Definition for a Node.
 * function Node(val, children) {
 *    this.val = val;
 *    this.children = children;
 * };
 */

/**
 * @param {Node|null} root
 * @return {number[]}
 */
var preorder = function (root) {
  if (!root) return [];
  let res = [];
  var preorderNode = (node) => {
    if (!node) return [];
    res.push(node.val);
    // n叉树注意这里
    for (child of node.children) {
      preorderNode(child);
    }
  };
  preorderNode(root);
  return res;
};
```
