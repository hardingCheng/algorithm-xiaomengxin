### 方法 1

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var topKFrequent = function (nums, k) {
  let map = new Map();
  let arr = new Set(...nums);
  nums.map((item, index) => {
    if (map.has(item)) {
      map.set(item, map.get(item) + 1);
    }else{
      map.set(item, 1))
    }
  });
  return arr.sort((a, b) => map.get(a) - map.get(b)).slice(0,k);
};
```
